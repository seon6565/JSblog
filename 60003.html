)COMMENT = '테이블코멘트'

COLLATE CHARSET 지정(utf8mb4_general_ci, utf8mb4_unicode_ci, ...등등)

INSERT INTO `테이블명`(칼럼명1,칼럼명2......) VALUES (값1,값2........)

DELETE FROM `테이블명` WHERE 조건

UPDATE `테이블명` SET `칼럼명`=값, `칼럼명2`=값..... WHERE 조건

SELECT 칼럼명 FROM 테이블명 WHERE 조건 GROUP BY 칼럼명 HAVING 칼럼명 = 조건(GROUP BY에 종속) ORDER BY 정렬조건 LIMIT 시작점, 갯수
*FROM 이후로는 필요에 따라서 추가해서 사용

SELECT DISTINCT 중복 제거해서 조회

SELECT 칼럼1 AS '칼럼명' 조회 시 칼럼의 이름 지정

SELECT * INTO OUTFILE '저장할 파일이름 포함한 경로'
FIELDS TERMINATED BY '칼럼간의 구분자추가' ENCLOSED BY '칼럼 내용의 처음과 끝에 추가' FROM 테이블명 파일로 저장하기

LOAD DATA LOCAL INFILE '파일 이름 포함한 경로'
INTO TABLE `테이블명` FILEDS TERMINATED BY '칼럼간구분자' ENCLOSED BY '칼럼 내용의 처음과 끝'

WHERE `칼럼명` LIKE '%와일드카드% %A A로 끝나는 모든 조건 A% A로 시작하는 모든 조건 %A% A가 포함된 모든 조건

WHERE `칼럼명` IN('1','3') IN 내부의 값이 있는 것들 TRUE 리턴하는 조건

WHERE `칼럼명` NOT = 조건 조건이 아닌 것들 TRUE 리턴하는 조건

WHERE IS NULL, WHERE IS NOT NULL NULL유무 확인하는 조건

산술연산자 +, -, *, DIV(몫), %(나머지)

:=, = SELECT문에서 할당 시 :=를 사용해야 하며, SET에서는 =로 할당 (이외에는 =는 비교연산자로 할당이 아님)

NOT, XOR NOT(연산에 대한 부정), XOR(하나만 TRUE)
